Backend
=======
# from project root (where backend/ folder sits)
python3 -m venv venv
source venv/bin/activate     # Linux / macOS
# OR venv\Scripts\activate    # Windows
pip install -r requirements.txt

# set env vars (or copy backend/.env.example -> backend/.env and edit)
export DB_USER=radius
export DB_PASS=radpass
export DB_HOST=localhost
export DB_NAME=radius

# start
python3 run.py        # app.listen -> http://127.0.0.1:5000
